SELECT d.dept_no, d.dept_name, COUNT(e.emp_no) AS num_employees
FROM departments d
JOIN dept_emp de ON d.dept_no = de.dept_no
JOIN employees e ON de.emp_no = e.emp_no
WHERE de.to_date = '9999-01-01'
GROUP BY d.dept_no, d.dept_name
